fig, axes = plt.subplots(2,2,figsize=(8,8))
leg = []
labels=[r"$\theta = 0$",r"$\theta = \pi/4$",r"$\theta = \pi/2$",r"$\theta = 3\pi/4$"]
for ii, theta_ in enumerate([0,np.pi/4,np.pi/2,np.pi*3/4]):
    syn = syn_theta[ii]
    ln, = axes[0,0].plot(wvl[:-1],syn[0,0,0][:-1],label=labels[ii])
    axes[0,1].plot(wvl[:-1],syn[0,0,1][:-1])
    axes[1,0].plot(wvl[:-1],syn[0,0,2][:-1])
    axes[1,1].plot(wvl[:-1],syn[0,0,3][:-1])
    leg.append(ln)

axes[0,0].set_ylabel(r"$I$",fontsize=12) 
axes[0,1].set_ylabel(r"$Q$",fontsize=12)
axes[1,0].set_ylabel(r"$U$",fontsize=12)
axes[1,1].set_ylabel(r"$V$",fontsize=12)

for ax_ in axes.flatten():
    ax_.tick_params(labelsize=12)
    ax_.set_xlabel("$\lambda - 6301\ (\mathrm{\AA})$",fontsize=12)
axes[0,1].legend(leg,[leg_.get_label() for leg_ in leg],loc = 'upper right',fontsize = 12)
plt.subplots_adjust(wspace=0.3)
fig.suptitle(r'$B=1000\ \mathrm{G},\ \varphi=\pi/8,\ \Delta \lambda = 0.05\ \mathrm{\AA}$', fontsize=12,y=0.92)
plt.savefig(fname = 'changetheta.pdf', dpi = 300, format = 'pdf')
